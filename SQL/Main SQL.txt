--------------------------------------------------
-- Table Creations
--------------------------------------------------

-- Speciality (Parent table for Wards and Consultants)
CREATE TABLE SPECIALITY (
    specialityName VARCHAR(100) PRIMARY KEY
);

-- Ward
CREATE TABLE WARD (
    wardName VARCHAR(100) PRIMARY KEY,
    specialityName VARCHAR(100),
    FOREIGN KEY (specialityName) REFERENCES SPECIALITY(specialityName)
);

-- CareUnit
CREATE TABLE CAREUNIT (
    careUnitID INT PRIMARY KEY,
    wardName VARCHAR(100),
    inChargeNurseID INT, 
    FOREIGN KEY (wardName) REFERENCES WARD(wardName)
);

-- Nurse 
CREATE TABLE NURSE (
    nurseID INT PRIMARY KEY,
    staffType VARCHAR(50),
    serviceTime TIME,
    careUnitID INT,
    FOREIGN KEY (careUnitID) REFERENCES CAREUNIT(careUnitID)
);

-- Adding circular dependency for CareUnit In-Charge
ALTER TABLE CAREUNIT ADD CONSTRAINT fk_nurse_incharge 
FOREIGN KEY (inChargeNurseID) REFERENCES NURSE(nurseID);

-- Bed
CREATE TABLE BED (
    bedNo INT PRIMARY KEY,
    careUnitID INT,
    FOREIGN KEY (careUnitID) REFERENCES CAREUNIT(careUnitID)
);

-- Patient
CREATE TABLE PATIENT (
    patientNo INT PRIMARY KEY,
    patientName VARCHAR(100),
    DOB DATE,
    careUnitID INT,
    FOREIGN KEY (careUnitID) REFERENCES CAREUNIT(careUnitID)
);

-- TeamOfDoctors
CREATE TABLE TEAMOFDOCTORS (
    SNo INT PRIMARY KEY,
    leaderDoctorNo INT 
);

-- Doctor 
CREATE TABLE DOCTOR (
    doctorNo INT PRIMARY KEY,
    doctorName VARCHAR(100),
    teamID INT,
    position VARCHAR(50),
    dateOfJoin DATE,
    FOREIGN KEY (teamID) REFERENCES TEAMOFDOCTORS(SNo)
);

-- Adding leader constraint back to Doctor
ALTER TABLE TEAMOFDOCTORS ADD CONSTRAINT fk_leader 
FOREIGN KEY (leaderDoctorNo) REFERENCES DOCTOR(doctorNo);

-- Consultant (Specialization of Doctor)
CREATE TABLE CONSULTANT (
    doctorNo INT PRIMARY KEY,
    specialityName VARCHAR(100),
    FOREIGN KEY (doctorNo) REFERENCES DOCTOR(doctorNo),
    FOREIGN KEY (specialityName) REFERENCES SPECIALITY(specialityName)
);

-- Doctor Experience 
CREATE TABLE DOCTOR_EXPERIENCE (
    doctorNo INT,
    fromDate DATE,
    toDate DATE,
    position VARCHAR(50),
    establishment VARCHAR(100),
    PRIMARY KEY (doctorNo, fromDate),
    FOREIGN KEY (doctorNo) REFERENCES DOCTOR(doctorNo)
);

-- Doctor Progress 
CREATE TABLE DOCTOR_PROGRESS (
    doctorNo INT,
    date DATE,
    grade VARCHAR(10),
    PRIMARY KEY (doctorNo, date),
    FOREIGN KEY (doctorNo) REFERENCES DOCTOR(doctorNo)
);

-- Complaint
CREATE TABLE COMPLAINT (
    complaintCode INT PRIMARY KEY,
    complaintType VARCHAR(100),
    patientNo INT,
    FOREIGN KEY (patientNo) REFERENCES PATIENT(patientNo)
);

-- Treatment
CREATE TABLE TREATMENT (
    treatmentCode INT PRIMARY KEY,
    startDate DATE,
    endDate DATE,
    complaintCode INT,
    FOREIGN KEY (complaintCode) REFERENCES COMPLAINT(complaintCode)
);

-- TreatmentDoneBy (M:N Relationship between Doctor and Treatment)
CREATE TABLE TREATMENTDONEBY (
    doctorID INT,
    treatmentCode INT,
    fromDate DATE,
    toDate DATE,
    PRIMARY KEY (doctorID, treatmentCode),
    FOREIGN KEY (doctorID) REFERENCES DOCTOR(doctorNo),
    FOREIGN KEY (treatmentCode) REFERENCES TREATMENT(treatmentCode)
);

-- Admitted (Linking Patient, Bed, and Consultant)
CREATE TABLE ADMITTED (
    patientNo INT,
    SNo INT,
    bedNo INT,
    dateAdmitted DATE,
    consultantDoctorNo INT,
    PRIMARY KEY (patientNo, SNo, bedNo),
    FOREIGN KEY (patientNo) REFERENCES PATIENT(patientNo),
    FOREIGN KEY (bedNo) REFERENCES BED(bedNo),
    FOREIGN KEY (consultantDoctorNo) REFERENCES CONSULTANT(doctorNo)
);
